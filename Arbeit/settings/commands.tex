\newcommand{\insertfigure}[3]{ %filename and label, width, caption
	\begin{figure}
		\center
		\includegraphics[width=#2\columnwidth]{images/#1}
		\vspace{-0.2cm}
		\caption{#3}
		\label{fig:#1}
	\end{figure}
}

\newcommand{\insertexample}[3] { %label, caption, text
	\begin{example-}
		\KOMAoptions{captions=nooneline}

	\end{example-}
}

	}
